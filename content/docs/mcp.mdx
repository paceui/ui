Paste the following code into your IDE to configure the Registry MCP.

<Tabs defaultValue="cursor" className="mt-5">
    <TabsList>
        <TabsTrigger value="cursor">Cursor</TabsTrigger>
        <TabsTrigger value="windsurf">Windsurf</TabsTrigger>
    </TabsList>
    <TabsContent value="cursor" className="mt-4">
        <CodePreview lang="json" code={`// Copy and paste the code into .cursor/mcp.json
{
    "mcpServers": {
      "shadcn": {
        "command": "npx",
        "args": ["-y", "shadcn@canary", "registry:mcp"],
        "env": {
          "REGISTRY_URL": "https://ui.paceui.com/r/mcp.json"
        }
      }
    }
}`}/>
    </TabsContent>
    <TabsContent value="windsurf" className="mt-4">
        <CodePreview lang="json" code={`// Copy and paste the code into .codeium/windsurf/mcp_config.json
{
    "mcpServers": {
      "shadcn": {
        "command": "npx",
        "args": ["-y", "shadcn@canary", "registry:mcp"],
        "env": {
          "REGISTRY_URL": "https://ui.paceui.com/r/mcp.json"
        }
      }
    }
}`}/>
    </TabsContent>
</Tabs>
